CREATION OF A DATABASE USING DDL COMMANDS
-----------------------------------------

SQL> desc Stud1;

 ----------------------------------------- -------- ----------------------------

SQL> create table cust(cust_id  number primary key,address varchar(25),city varchar(20),ph_no number);

Table created.

SQL> desc cust
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 CUST_ID                                   NOT NULL NUMBER
 ADDRESS                                            VARCHAR2(25)
 CITY                                               VARCHAR2(20)
 PH_NO                                              NUMBER

SQL> alter table cust add name varchar(15);

Table altered.

SQL> desc cust;
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 CUST_ID                                   NOT NULL NUMBER
 ADDRESS                                            VARCHAR2(25)
 CITY                                               VARCHAR2(20)
 PH_NO                                              NUMBER
 NAME                                               VARCHAR2(15)



SQL> create table product1(p_id number primary key,pname varchar(20),price number default 0);

Table created.

SQL> desc product1;
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 P_ID                                      NOT NULL NUMBER
 PNAME                                              VARCHAR2(20)
 PRICE                                              NUMBER



SQL> create  table invoice_master (inv_id int,cust_id int,inv_date date,constraint fk_cust_id foreign key(cust_id)references cust(cust_id));

Table created.

SQL> desc invoice_master;
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 INV_ID                                             NUMBER(38)
 CUST_ID                                            NUMBER(38)
 INV_DATE                                           DATE


SQL> alter table invoice_master modify inv_id int primary key;

Table altered.

SQL> desc invoice_master;
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 INV_ID                                    NOT NULL NUMBER(38)
 CUST_ID                                            NUMBER(38)
 INV_DATE                                           DATE

SQL> create  table invoice_item(inv_id int,constraint inv_id foreign key(inv_id)references invoice_master(inv_id),p_id int,constraint p_id foreign key(p_id)references product1(p_id),quantity int);

Table created.


SQL> desc invoice_item;
 Name                                      Null?    Type
 ----------------------------------------- -------- ----------------------------
 INV_ID                                             NUMBER(38)
 P_ID                                               NUMBER(38)
 QUANTITY                                           NUMBER(38)
