DECLARE
   N NUMBER:=&N;
   REV NUMBER:=0;
   REM NUMBER;
   TEMP NUMBER:=N;
BEGIN
   WHILE N>0 LOOP
      REM:=MOD(N,10);
      REV:=(REV*10)+REM;
      N:=TRUNC(N/10);
   END LOOP;
IF TEMP=REV THEN
    DBMS_OUTPUT.PUT_LINE(TEMP||' IS PALINDROME');
ELSE
    DBMS_OUTPUT.PUT_LINE(TEMP||' IS NOT PALINDROME');
END IF;
END;
